\hypertarget{md_README_autotoc_md1}{}\doxysection{Algorithm}\label{md_README_autotoc_md1}

\begin{DoxyEnumerate}
\item {\bfseries{Initialization}}\+:
\begin{DoxyItemize}
\item Import necessary libraries.
\item Initialize variables for the number of buses {\ttfamily n} and the choice of input type (impedance or admittance).
\end{DoxyItemize}
\item {\bfseries{Input Number of Buses}}\+:
\begin{DoxyItemize}
\item Prompt the user to enter the number of buses.
\item Initialize the bus admittance matrix {\ttfamily Ybus} with complex zeros.
\item Initialize a placeholder matrix {\ttfamily y} with complex zeros.
\end{DoxyItemize}
\item {\bfseries{Input Choice for Impedance or Admittance}}\+:
\begin{DoxyItemize}
\item Prompt the user to choose between entering impedance or admittance values.
\end{DoxyItemize}
\item {\bfseries{Input Impedance or Admittance Values}}\+:
\begin{DoxyItemize}
\item {\bfseries{Case 1}}\+: If the user chooses impedance\+:
\begin{DoxyItemize}
\item For each pair of buses, prompt the user to enter the impedance value.
\item Calculate the admittance for each pair as the reciprocal of the impedance and store it in matrix {\ttfamily y}.
\end{DoxyItemize}
\item {\bfseries{Case 2}}\+: If the user chooses admittance\+:
\begin{DoxyItemize}
\item For each pair of buses, prompt the user to enter the admittance value directly and store it in matrix {\ttfamily y}.
\end{DoxyItemize}
\end{DoxyItemize}
\item {\bfseries{Form Bus Admittance Matrix}}\+:
\begin{DoxyItemize}
\item For each bus {\ttfamily i}\+:
\begin{DoxyItemize}
\item For each bus {\ttfamily j}\+:
\begin{DoxyItemize}
\item If {\ttfamily i} equals {\ttfamily j} (diagonal elements)\+:
\begin{DoxyItemize}
\item Sum the admittances of all lines connected to bus {\ttfamily i} and update {\ttfamily Ybus\mbox{[}i\mbox{]}\mbox{[}j\mbox{]}}.
\end{DoxyItemize}
\item Otherwise (off-\/diagonal elements)\+:
\begin{DoxyItemize}
\item Set the element {\ttfamily Ybus\mbox{[}i\mbox{]}\mbox{[}j\mbox{]}} to the negative of the corresponding admittance value {\ttfamily y\mbox{[}i\mbox{]}\mbox{[}j\mbox{]}}.
\end{DoxyItemize}
\end{DoxyItemize}
\end{DoxyItemize}
\end{DoxyItemize}
\item {\bfseries{Output Bus Admittance Matrix}}\+:
\begin{DoxyItemize}
\item Print the bus admittance matrix {\ttfamily Ybus}.
\end{DoxyItemize}
